---
title: Monitoring Metrics
description: List of metrics Keywords API logs keep track of
---

<ResponseField name="prompt_tokens" type="integer">
Number of tokens in the prompt. Default is 0.
</ResponseField>

<ResponseField name="completion_tokens" type="integer">
Number of tokens in the completion. Default is 0.
</ResponseField>

<ResponseField name="cost" type="float">
Cost of the operation. Default is 0.
</ResponseField>

<ResponseField name="prompt_messages" type="array">
Array of prompt messages. Default is an empty list.
</ResponseField>

<ResponseField name="completion_message" type="dict">
Completion message in JSON format. Default is an empty dictionary.
</ResponseField>

<ResponseField name="latency" type="float">
Latency of the operation in seconds. Default is 0.
</ResponseField>

<ResponseField name="scores" type="dict">
Scores in JSON format. Default is an empty dictionary.
</ResponseField>

<ResponseField name="user" type="foreign key">
Foreign key to the User model.
</ResponseField>

<ResponseField name="organization_key" type="foreign key">
Foreign key to the OrganizationKey model. Default is the default organization key.
</ResponseField>

<ResponseField name="model" type="string">
Model used for the operation. Default is an empty string.
</ResponseField>

<ResponseField name="timestamp" type="datetime">
Timestamp of the operation. Default is the current time.
</ResponseField>

<ResponseField name="failed" type="boolean">
Whether the operation failed. Default is false.
</ResponseField>

<ResponseField name="error_message" type="text">
Error message if the operation failed. Default is an empty string.
</ResponseField>

<ResponseField name="category" type="string">
Category of the operation. Default is an empty string.
</ResponseField>

<ResponseField name="time_to_first_token" type="float">
Time to first token in seconds. Default is 0.
</ResponseField>

<ResponseField name="routing_time" type="float">
Routing time in seconds. Default is 0.
</ResponseField>

<ResponseField name="sentiment_analysis" type="dict">
Sentiment analysis in JSON format. Default is an empty dictionary.
</ResponseField>

<ResponseField name="metadata" type="dict">
Metadata in JSON format. Default is an empty dictionary.
</ResponseField>

<ResponseField name="stream" type="boolean">
Whether the operation was streamed. Default is false.
</ResponseField>

<ResponseField name="error_code" type="integer">
Error code if the operation failed. Default is 0.
</ResponseField>

<ResponseField name="customer_identifier" type="string">
Customer identifier. Default is an empty string.
</ResponseField>

<ResponseField name="cached_response" type="integer">
Whether the response was cached. Default is 0.
</ResponseField>
